) -> WinResult<String>

IFileSinkFilter::GetCurFile method.

If you pass an AM_MEDIA_TYPE reference to pmt, its pbFormat field may return a valid reference to a format block. If so, you must free it with CoTaskMemFree, or you’ll have a memory leak.

Examples

ⓘ
use winsafe::prelude::*;
use winsafe::{CoMemTaskFree, dshow};

let isink: dshow::IFileSinkFilter; // initialized somewhere

let mut ammt = dshow::AM_MEDIA_TYPE::default();
unsafe {
    isink.GetCurFile(Some(&mut ammt))?;
    if let Some(pb_format) = ammt.pbFormat::<dshow::DVINFO>() {
        CoTaskMemFree(pb_format);
    }
}
