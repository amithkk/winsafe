fn get_Height(&self) -> HrResult<i32>

IPicture::get_Height method.

Note: Returns a value in HIMETRIC units. Use HDC::HiMetricToPixel to convert it to pixels.

Examples
ⓘ
use winsafe::prelude::*;
use winsafe::{HWND, idl};

let pic: idl::IPicture; // initialized somewhere

let hdc = HWND::NULL.GetDC()?;

let (_, height) = hdc.HiMetricToPixel(0, pic.get_Height()?);
println!("Height: {} px", height);

HWND::NULL.ReleaseDC(hdc)?;
