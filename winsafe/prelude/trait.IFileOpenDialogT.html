use winsafe::prelude::*;

let fo: shell::IFileOpenDialog; // initialized somewhere

let paths = fo.GetResults()?.iter()
    .map(|shi|
        shi.and_then(|shi|
            shi.GetDisplayName(SIGDN::FILESYSPATH)
        )
    )
    .collect::<WinResult<Vec<_>>>()?,
